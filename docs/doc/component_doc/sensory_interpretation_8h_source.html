359 #define NUMBER_OF_PRIMITIVES 25 // we allow for different event rates for each perception primitive being simulated
360  // (see Deliverable D1.3)
361  // this makes it easier to check that the simulation is work by setting the event rate
362  // of all but one of the primitives to zero, periodically changing the selected primitive
363  // every 10 seconds or so, for example
364 
365 #define NUMBER_OF_JOINTS 6 // the number of of joint angles describing the childs current body pose.
366  // The joint angles correspond to a canonical representation of a body skeleton,
367  // yet to be defined (see D1.3, getBodyPose() perception primitive)
368 
369 #define NUMBER_OF_FACES 2 // the number of faces that can be returned by the getFaces() primitive.
370  // The primitive determines the locations where faces are visible
371  // in the therapy environment (viewed from any of the therapy cameras).
372  // Each location is specified by a triple of Cartesian coordinates x, y, and z.
373 
374 #define NUMBER_OF_EXPRESSIONS 4 // the number of face expressions that can be returned by the getFacesExpression() primitive.
375  // The primitive determines the locations where faces are visible
376  // in the therapy environment (viewed from any of the therapy cameras).
377  // Each location is specified by a triple of Cartesian coordinates x, y, and z.
378 
379 #define NUMBER_OF_HANDS 2 // the number of hands that can be returned by the getHands() primitive.
380  // The primitive determines the locations where hands are visible
381  // in the therapy environment (viewed from any of the therapy cameras).
382  // Each location is specified by a triple of Cartesian coordinates x, y, and z.
383 
384 #define NUMBER_OF_VOICES 2 // the cardinality of the voice class that can be returned by the identifyVoice() primitive.
385 
386 #define NUMBER_OF_OBJECTS 2 // the number of objects that can be returned by the getObjects() primitive.
387  // The primitive determines the locations where objects are visible
388  // in the therapy environment (viewed from any of the therapy cameras).
389  // Each location is specified by a triple of Cartesian coordinates x, y, and z.
390 
391 #define NUMBER_OF_TRAJECTORIES 4 // the cardinality of the trajectory class that can be returned by the identifyTrajectory() primitive.
392  // the trajectory defined by a vector of 4-tuples, each 4-tuple defining a 3-D location and a time-stamp.
393  // The possible classes are determined during the set up phase. Typically, they will correspond to
394  // a designated set of hand gestures to be made by the child (e.g. a wave).
395
